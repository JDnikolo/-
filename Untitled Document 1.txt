MongoDB Setup:

install docker
vue-cli


docker run -d --name base -v /data/db -p 27017:27017 mongo

db import/export:
mongoimport -d local -c Studies /tmp/Studies.json
mongoexport -d local -c Studies --out=/tmp/Studies.json
copy from container to host:
docker cp base:/tmp/Studies.json /tmp/Studies.json

access mongo docker
docker exec -it base /bin/bash
